// 固定長配列
int arr[3] = {1, 2, 3};

// std::vector（可変長配列）
#include <vector>

3-2 mapとイテレーター

#include <map>
std::map<std::string, int> score;
score["Alice"] = 90;
score["Bob"] = 80;

// イテレータで全要素表示
for (auto it = score.begin(); it != score.end(); ++it) {
  std::cout << it->first << ": " << it->second << std::endl;
}
